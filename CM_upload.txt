def compute_p_values(data):
    """Compute Pearson correlation p-values with symmetry optimization"""
    n_vars = data.shape[1]
    p_values = np.zeros((n_vars, n_vars))
    col_names = data.columns
    
    for i in range(n_vars):
        for j in range(i, n_vars):
            _, p_val = pearsonr(data.iloc[:, i], data.iloc[:, j])
            p_values[i, j] = p_val
            p_values[j, i] = p_val
    
    return pd.DataFrame(p_values, columns=col_names, index=col_names)

def create_diverging_colormap():
    """Create a colorblind-friendly diverging colormap (blue to red)"""
    colors = ["#08306b", "#08519c", "#2171b5", "#4292c6", 
              "#6baed6", "#9ecae1", "#c6dbef", "#eff3ff",
              "#f7f7f7",  # Center (white)

              "#ef3b2c", "#cb181d", "#a50f15", "#67000d"]
    return LinearSegmentedColormap.from_list("enhanced_diverging", colors, N=256)

def create_sequential_colormap():

              "#969696", "#737373", "#525252", "#252525",

def significance_level(p):
    """Return significance symbols with asterisks"""
    if p <= 0.001: return '***'
    if p <= 0.01:  return '**'
    if p <= 0.05:  return '*'
    return ''

def apply_enhanced_style(ax, title, cbar_label):
    """Apply enhanced publication styling with black borders"""
    # Title styling - BIGGER AND BOLDER
    ax.set_title(title, fontsize=22, fontweight='black', pad=25, color='black')
    cbar = ax.collections[0].colorbar
    cbar.set_label(cbar_label, fontsize=16, fontweight='black', labelpad=15)
    cbar.ax.tick_params(labelsize=14, width=2, colors='black')
    cbar.outline.set_linewidth(3)
    cbar.outline.set_edgecolor('black')
    plt.setp(ax.get_xticklabels(), rotation=45, ha='right',

    for spine in ax.spines.values():
        spine.set_linewidth(4)
        spine.set_color('black')

def add_enhanced_annotations(ax, corr_matrix, p_value_matrix, color_scheme='color'):
    """Add LARGER, BOLDER correlation values and significance stars"""
    n_rows, n_cols = corr_matrix.shape
    for i in range(n_rows):
        for j in range(n_cols):
            corr = corr_matrix.iloc[i, j]
            pval = p_value_matrix.iloc[i, j]
            sig = significance_level(pval)
            if color_scheme == 'color':
                # Enhanced contrast for better visibility
                if abs(corr) > 0.8:
                    txt_color = 'white'
                    txt_weight = 'black'  # Extra bold for high contrast
                elif abs(corr) > 0.5:
                    txt_color = 'white'
                    txt_weight = 'bold'
                else:
                    txt_color = 'black'
                    txt_weight = 'bold'
            else:
                # For grayscale
                if abs(corr) > 0.7:
                    txt_color = 'white'
                    txt_weight = 'black'
                elif abs(corr) > 0.4:
                    txt_color = 'white'
                    txt_weight = 'bold'
                else:
                    txt_color = 'black'
                    txt_weight = 'bold'
            ax.text(
                j + 0.5, i + 0.65,
                f"{corr:.2f}",
                ha="center", va="center",
                fontsize=15,  # Increased from 11
                fontweight=txt_weight,
                color=txt_color,
                bbox=dict(boxstyle="round,pad=0.2", 
                         facecolor='none' if abs(corr) < 0.3 else 'black',
                         alpha=0.1 if abs(corr) < 0.3 else 0,
                         edgecolor='none')
            )

            if sig:
                ax.text(
                    j + 0.5, i + 0.35,
                    sig,
                    ha="center", va="center",
                    fontsize=14,  # Increased from 10
                    fontweight='black',
                    color=txt_color
                )
