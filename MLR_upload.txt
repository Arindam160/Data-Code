import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import statsmodels.api as sm
from sklearn.metrics import r2_score, mean_squared_error, mean_absolute_error
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split


file_path = ""
output_dir = ""
os.makedirs(output_dir, exist_ok=True)  # Create folder if not exists
data = pd.read_csv(file_path)
data.columns = data.columns.str.strip()
X_columns = []
Y_column = ''

if not set(X_columns + [Y_column]).issubset(data.columns):
    missing_cols = set(X_columns + [Y_column]) - set(data.columns)
    raise ValueError(f" Missing columns: {missing_cols}")

X = data[X_columns]
Y = data[Y_column]
X_train, X_test, Y_train, Y_test = train_test_split(X, Y, test_size=0.3, random_state=42)
scaler = StandardScaler()
X_train_scaled = scaler.fit_transform(X_train)
X_test_scaled = scaler.transform(X_test)
X_train_scaled = pd.DataFrame(X_train_scaled, columns=X_columns, index=X_train.index)
X_test_scaled = pd.DataFrame(X_test_scaled, columns=X_columns, index=X_test.index)
X_train_sm = sm.add_constant(X_train_scaled)
X_test_sm = sm.add_constant(X_test_scaled)
model_sm = sm.OLS(Y_train, X_train_sm).fit()
Y_train_pred = model_sm.predict(X_train_sm)
Y_test_pred = model_sm.predict(X_test_sm)
X_scaled_full = scaler.transform(X)
X_scaled_full = pd.DataFrame(X_scaled_full, columns=X_columns, index=X.index)
X_sm_full = sm.add_constant(X_scaled_full)
Y_pred_full = model_sm.predict(X_sm_full)

def calc_metrics(y_true, y_pred):
    r2 = r2_score(y_true, y_pred) * 100
    mse = mean_squared_error(y_true, y_pred)
    rmse = np.sqrt(mse)
    mae = mean_absolute_error(y_true, y_pred)
    return r2, mse, rmse, mae

r2_train, mse_train, rmse_train, mae_train = calc_metrics(Y_train, Y_train_pred)
r2_test, mse_test, rmse_test, mae_test = calc_metrics(Y_test, Y_test_pred)
r2_full, mse_full, rmse_full, mae_full = calc_metrics(Y, Y_pred_full)

coefficients = pd.DataFrame({
    'Variable': X_train_sm.columns,
    'Coefficient': model_sm.params.round(6),
    'Standard Error': model_sm.bse.round(6),
    't-Statistic': model_sm.tvalues.round(6),
    'P-Value': model_sm.pvalues.round(6)
})

coefficients_output_path = os.path.join(output_dir, 'MLR_Coefficients.csv')
coefficients.to_csv(coefficients_output_path, index=False)
print(f"Coefficients and p-values saved to: {coefficients_output_path}")

p_values_df = coefficients[['Variable', 'P-Value']]
p_values_output_path = os.path.join(output_dir, 'Significance_Values.csv')
p_values_df.to_csv(p_values_output_path, index=False)
print(f"Significance values (p-values) saved to: {p_values_output_path}")

figsize_main = (12, 8)
figsize_bw = (12, 8)

scatter_size_train = 80
scatter_size_test = 80
scatter_size_bw = 90

line_width_ideal = 6
line_width_ideal_bw = 6

line_style_ideal = '-'
line_style_ideal_bw = '--'

color_train = 'blue'
color_test = 'orange'
color_ideal = 'green'
color_ideal_bw = 'black'

title_color = 'red'
title_fontweight = 'bold'

xlabel = 'Observed'
ylabel = 'Predicted'

grid_linewidth = 0.0
grid_linestyle = ''
grid_color = 'white'

legend_fontsize = 18
label_fontsize = 18
title_fontsize = 24

# Plot observed vs. predicted values
plt.figure(figsize=figsize_main)
sns.scatterplot(x=Y_train, y=Y_train_pred, color=color_train, label='Train: Observed vs Predicted', s=scatter_size_train)
sns.scatterplot(x=Y_test, y=Y_test_pred, color=color_test, label='Test: Observed vs Predicted', s=scatter_size_test)
plt.plot([Y.min(), Y.max()], [Y.min(), Y.max()], color=color_ideal, linestyle=line_style_ideal, linewidth=line_width_ideal, label='Ideal Fit')
plt.xlabel(xlabel, fontsize=label_fontsize)
plt.ylabel(ylabel, fontsize=label_fontsize)
plt.title(
    f'',
    color=title_color, fontweight=title_fontweight, fontsize=title_fontsize
)
plt.legend(fontsize=legend_fontsize)
plt.grid(True, linestyle=grid_linestyle, linewidth=grid_linewidth, color=grid_color)
plt.xticks(fontsize=label_fontsize)
plt.yticks(fontsize=label_fontsize)
plt.show()

intercept = model_sm.params[0]
coefs = model_sm.params[1:]

# Construct regression equation
equation_terms = [f"({coef:.6f} * {var})" for coef, var in zip(coefs, X_train_sm.columns[1:])]
regression_equation = f"Y = {intercept:.6f} + " + " + ".join(equation_terms)
print("\n**Regression Equation:**")
print(regression_equation)
positive_sensitivity = coefs[coefs > 0].sort_values(ascending=False)
negative_sensitivity = coefs[coefs < 0].sort_values()

print("\nPositive Sensitivity Variables (Higher values increase Hydrogen Abs. Time):")
for var, coef in positive_sensitivity.items():
    print(f"  {var}: {coef:.6f}")

print("\n Negative Sensitivity Variables (Higher values decrease Hydrogen Abs. Time):")
for var, coef in negative_sensitivity.items():
    print(f"  {var}: {coef:.6f}")

# Save results into CSV files in the output directory
positive_sensitivity_df = pd.DataFrame({
    'Variable': positive_sensitivity.index,
    'Coefficient': positive_sensitivity.values
})
positive_sensitivity_df.to_csv(os.path.join(output_dir, "positive_sensitivity.csv"), index=False)

negative_sensitivity_df = pd.DataFrame({
    'Variable': negative_sensitivity.index,
    'Coefficient': negative_sensitivity.values
})
negative_sensitivity_df.to_csv(os.path.join(output_dir, "negative_sensitivity.csv"), index=False)
with open(os.path.join(output_dir, "regression_equation.txt"), "w") as f:
    f.write("Multiple Linear Regression Equation:\n")
    f.write("====================================\n\n")
    f.write(regression_equation)
    f.write("\n\nWhere Y is the predicted")
    f.write("\n\nNote: All independent variables are standardized (z-scores)")

print(f"\n Files saved in {output_dir}:")
print("   - regression_equation.txt")
print("   - positive_sensitivity.csv")
print("   - negative_sensitivity.csv")

# Print key metrics
print("\n--- Regression Results ---")
print(f" Training Set: R² = {r2_train:.2f}%, MSE = {mse_train:.6f}, RMSE = {rmse_train:.6f}, MAE = {mae_train:.6f}")
print(f" Test Set: R² = {r2_test:.2f}%, MSE = {mse_test:.6f}, RMSE = {rmse_test:.6f}, MAE = {mae_test:.6f}")
print(f" Overall: R² = {r2_full:.2f}%, MSE = {mse_full:.6f}, RMSE = {rmse_full:.6f}, MAE = {mae_full:.6f}")

# Print coefficients with p-values
print("\n---  Coefficients and P-Values (Training Model) ---")
print(coefficients.to_string(index=False))

# Determine model quality based on test R²
if r2_test > 75:
    print("\n\033[1;32m The model has a good fit on the test set.\033[0m")  # Green text for good fit
elif r2_test > 50:
    print("\n\033[1;33m The model has a moderate fit on the test set.\033[0m")  # Yellow for moderate fit
else:
    print("\n\033[1;31m The model has a poor fit on the test set. Check the regression equation.\033[0m")  # Red for poor fit
